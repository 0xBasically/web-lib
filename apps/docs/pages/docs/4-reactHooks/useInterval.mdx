# useInterval
This hook will transform setInterval to a proper React hook [react-recipe](https://github.com/craig1123/react-recipes/blob/master/docs/useInterval.md) repository.

```tsx
import {useInterval} from '@yearn-finance/web-lib/hooks';
```

## Usage

```tsx
import {useInterval} from '@yearn-finance/web-lib/hooks';

function	App() {
	useInterval((): void => {
		doSomething();
	}, 7500, true, []);
}
```

## Return Value
*NA*


## Configuration

### `callback`
The function to trigger as callback, as you would do it with a standard `useEffect`.
```tsx {5}
import {useInterval} from '@yearn-finance/web-lib/hooks';

function	App() {
	useInterval(
		(): void => doSomething(),
		7500,
		true,
		[]
	);
}
```

### `delay`
The delay in milliseconds to wait between each callback calls. If not provided, the callback is not re-triggered.  
Default value is `1000`.
```tsx {6}
import {useInterval} from '@yearn-finance/web-lib/hooks';

function	App() {
	useInterval(
		(): void => doSomething(),
		7500,
		true,
		[]
	);
}
```

### `runOnLoad`
Indicate if the callback should be triggered on component mount. Default value is `false`.
```tsx {7}
import {useInterval} from '@yearn-finance/web-lib/hooks';

function	App() {
	useInterval(
		(): void => doSomething(),
		7500,
		true,
		[]
	);
}
```

### `effectDependencies`
Just like the `useEffect` hook, you can pass an array of dependencies to re-trigger the callback on changes. If one of the dependencies changes, the previous interval is cleared and the new callback is triggered.
```tsx {8}
import {useInterval} from '@yearn-finance/web-lib/hooks';

function	App(a: number, b: string, c: boolean) {
	useInterval(
		(): void => doSomething(),
		7500,
		true,
		[a, b, c]
	);
}
```
